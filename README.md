# review-nestjs-graphql

Beginner go here https://github.com/lightzane/learn-nestjs-graphql

Reviewing Nestjs GraphQL with resolving child fields

**question.resolver.ts**

```ts
@Resolver(() => Question) // Question is needed here to resolve child fields
export class QuestionResolver {

    constructor(private readonly questionService: QuestionService) { }

    @Query(() => Question)
    findQuestion(@Args() findQuestionArgs: FindQuestionArgs): Question {
        return this.questionService.findQuestion(findQuestionArgs);
    }

    // Child field
    @ResolveField(() => [Answer], { name: 'answers' })
    filterAnswers(@Parent() question: Question, @Args() filterAnswerArgs: FilterAnswerArgs): Answer[] {
        return this.questionService.filterAnswer(question, filterAnswerArgs.identifier);
    }

}
```

**Query 1** 
```graphql
{
  findQuestion(_id:"62c9436ce47e123c4fe353cf") {
    _id
    value
    answers {
      identifier
      value
      score
    }
  }
}
```

**Query 2**
```graphql
{
  findQuestion(_id:"62c9436ce47e123c4fe353cf") {
    _id
    value
    answers(identifier: "636e7af2-62ad-a697-acd0-6fe5928dd2c4") {
      identifier
      value
      score
    }
  }
}
```
